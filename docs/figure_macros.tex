% Shared print-friendly figure macros for the exoplanet report.
% RV and orbit panels are drawn natively with PGFPlots from CSV tables
% produced by examples/export_paper_figures.py.

\usepackage{pgfplots}
\pgfplotsset{compat=1.18}

\pgfplotsset{
  PaperRVAxisStyle/.style={
    width=0.95\linewidth,
    height=0.72\linewidth,
    xlabel={Time (days)},
    ylabel={Stellar radial velocity (m/s)},
    title={Radial Velocity Fit},
    grid=both,
    grid style={line width=0.2pt, gray!40},
    tick label style={font=\footnotesize},
    label style={font=\small},
    title style={font=\small\bfseries},
    legend style={
      font=\scriptsize,
      at={(0.98,0.98)},
      anchor=north east,
      draw=none,
      fill=white,
      fill opacity=0.85,
      text opacity=1,
    },
    legend cell align=left,
  },
  PaperOrbitAxisStyle/.style={
    width=0.95\linewidth,
    height=0.72\linewidth,
    xlabel={$x$ (AU)},
    ylabel={$z$ (AU)},
    title={Reflex Orbit (Velocity Verlet)},
    axis equal image,
    grid=both,
    grid style={line width=0.2pt, gray!40},
    tick label style={font=\footnotesize},
    label style={font=\small},
    title style={font=\small\bfseries},
    legend style={
      font=\scriptsize,
      at={(0.98,0.98)},
      anchor=north east,
      draw=none,
      fill=white,
      fill opacity=0.85,
      text opacity=1,
    },
    legend cell align=left,
  },
}

\newcommand{\PaperFitFigure}[3]{%
  % #1 stem, #2 caption, #3 label
  \begin{figure}[H]
  \centering
  \begin{minipage}[t]{0.49\textwidth}
  \centering
  \begin{tikzpicture}
  \begin{axis}[PaperRVAxisStyle]
  \addplot[
    only marks,
    mark=*,
    mark size=1.05pt,
    color=black,
    error bars/.cd,
    y dir=both,
    y explicit,
  ] table[
    col sep=comma,
    header=true,
    x=time_days,
    y=radial_velocity_ms,
    y error=uncertainty_ms,
  ] {figures/plotdata/#1_rv_obs.csv};
  \addlegendentry{observations}
  \addplot[
    no marks,
    thick,
    color=blue!70!black,
    each nth point=4,
    filter discard warning=false,
  ] table[
    col sep=comma,
    header=true,
    x=time_days,
    y=model_velocity_ms,
  ] {figures/plotdata/#1_rv_model.csv};
  \addlegendentry{best fit}
  \end{axis}
  \end{tikzpicture}
  \end{minipage}\hfill
  \begin{minipage}[t]{0.49\textwidth}
  \centering
  \begin{tikzpicture}
  \begin{axis}[PaperOrbitAxisStyle]
  \addplot[
    no marks,
    thick,
    color=blue!70!black,
  ] table[
    col sep=comma,
    header=true,
    x=planet_x_au,
    y=planet_z_au,
  ] {figures/plotdata/#1_orbit.csv};
  \addlegendentry{planet}
  \addplot[
    no marks,
    thick,
    color=orange!80!black,
  ] table[
    col sep=comma,
    header=true,
    x=star_x_au,
    y=star_z_au,
  ] {figures/plotdata/#1_orbit.csv};
  \addlegendentry{star}
  \end{axis}
  \end{tikzpicture}
  \end{minipage}
  \caption{#2}
  \label{#3}
  \end{figure}%
}
